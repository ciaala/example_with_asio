\begin{frame}
  \frametitle{Two Patterns for Asynchronous I/O}
  \begin{itemize}
    \item{Reactor}\pause{}
      \begin{itemize}
        \item{Notify when I/O is ready}
        \item{``Consumer'' must perform I/O}
      \end{itemize}\pause{}
    \item{Proactor}\pause{}
      \begin{itemize}
        \item{Notify when I/O is done}
        \item{``Consumer'' works with data}
      \end{itemize}\pause{}
  \end{itemize}
\end{frame}
